server {
    listen 80;
    server_name yourdomain.com;
    # Redirect all HTTP to HTTPS
    return 301 https://$host;
}

server {
    listen 443 ssl http2;
    server_name yourdomain.com;

    # SSL Parameters (Let's Encrypt paths)
    ssl_certificate /etc/letsencrypt/live/yourdomain.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/yourdomain.com/privkey.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;

    # HSTS (Strict-Transport-Security) - Enforced at Proxy Layer
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    # Root directory for React Static Files
    # UPDATE THIS PATH to your actual build directory on the server
    root /var/www/burnout-app/frontend/build;
    index index.html;

    # Serve React Frontend
    location / {
        try_files  / /index.html;
    }

    # Proxy API to Express
    location /api {
        proxy_pass http://127.0.0.1:4000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade ;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host ;
        proxy_set_header X-Real-IP ;
        proxy_set_header X-Forwarded-For ;
        proxy_set_header X-Forwarded-Proto ;
    }
}
